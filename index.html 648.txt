let cView='team', cPeriod='YTD', cTab='summary';
const chartReg = {};

function wTotal(w, e) {
  const arr = w.data[e] || [];
  return arr.reduce((a,b)=>a+b, 0);
}
function wDays(w, e) {
  const arr = w.data[e] || [];
  return arr.filter(v=>v>0).length;
}
function filteredWeeks() {
  if (cPeriod==='YTD') return WEEKS;
  return WEEKS.filter(w=>w.id===cPeriod);
}
function empStats(e) {
  let total=0, days=0;
  filteredWeeks().forEach(w=>{ total+=wTotal(w,e); days+=wDays(w,e); });
  return { total, days, avg: days>0 ? total/days : 0 };
}
function sName(n) { const p=n.split(' '); return p[0][0]+'. '+p[1]; }

function setView(v) {
  cView=v;
  document.querySelectorAll('.vbtn').forEach(b=>{
    const t=b.textContent.trim();
    b.classList.toggle('active', v==='team'?t==='Full Team':t===sName(v));
  });
  renderCurrent();
}
function setMonth(m) {
  cPeriod=m;
  document.querySelectorAll('.mbtn').forEach(b=>b.classList.toggle('active',
    (m==='YTD'&&b.textContent.trim()==='YTD')||(m==='w1'&&b.textContent.includes('4'))||(m==='w2'&&b.textContent.includes('11'))));
  renderCurrent();
}
function setTab(t) {
  cTab=t;
  document.querySelectorAll('.tab').forEach(b=>b.classList.toggle('active', b.getAttribute('onclick').includes("'"+t+"'")));
  document.querySelectorAll('[id^="p-"]').forEach(el=>el.classList.add('hidden'));
  document.getElementById('p-'+t).classList.remove('hidden');
  renderCurrent();
}

function destroyChart(id) {
  if (chartReg[id]) { try { chartReg[id].destroy(); } catch(e){} delete chartReg[id]; }
}

function renderCurrent() {
  switch(cTab) {
    case 'summary':    renderSummary(); break;
    case 'weekly':     renderWeekly(); break;
    case 'leaderboard':renderLeaderboard(); break;
    case 'threshold':  renderThreshold(); break;
    case 'charts':     renderCharts(); break;
    case 'notes':      renderNotes(); break;
  }
}

// ── SUMMARY ──────────────────────────────────────────────
function renderSummary() {
  const el = document.getElementById('p-summary');
  let best='', bestV=0, topAvgE='', topAvg=0;
  STAFF.forEach(e=>{ const s=empStats(e); if(s.total>bestV){bestV=s.total;best=e;} if(s.avg>topAvg){topAvg=s.avg;topAvgE=e;} });
  const teamTotal = STAFF.reduce((a,e)=>a+empStats(e).total, 0);

  if (cView==='team') {
    const sorted = [...STAFF].sort((a,b)=>empStats(b).total-empStats(a).total);
    const rows = sorted.map(e=>{
      const s=empStats(e);
      return `<tr><td>${e}</td><td style="font-weight:700;color:#0d9488">${s.total}</td><td>${s.avg.toFixed(1)}</td><td>${s.days}</td></tr>`;
    }).join('');
    el.innerHTML = `<div class="sg">
      <div class="sc t"><div class="lbl">Team Total</div><div class="val">${teamTotal.toLocaleString()}</div><div class="sub">Claims modified</div></div>
      <div class="sc a"><div class="lbl">Top Producer</div><div class="val">${best.split(' ')[1]}</div><div class="sub">${bestV} claims</div></div>
      <div class="sc b"><div class="lbl">Best Daily Avg</div><div class="val">${topAvg.toFixed(1)}</div><div class="sub">${topAvgE.split(' ')[0]}</div></div>
      <div class="sc p"><div class="lbl">Active Staff</div><div class="val">${STAFF.length}</div><div class="sub">Tracked employees</div></div>
    </div>
    <div class="st">Team Overview</div>
    <table><thead><tr><th>Employee</th><th>Total Modified</th><th>Avg/Day</th><th>Days Worked</th></tr></thead><tbody>${rows}</tbody></table>`;
  } else {
    const e=cView, s=empStats(e);
    const rows = filteredWeeks().map(w=>{
      const t=wTotal(w,e), d=wDays(w,e);
      return `<tr><td>${w.label}</td><td style="color:#0d9488;font-weight:700">${t}</td><td>${d}</td><td>${d>0?(t/d).toFixed(1):'-'}</td></tr>`;
    }).join('');
    el.innerHTML = `<div class="sg">
      <div class="sc t"><div class="lbl">Total Modified</div><div class="val">${s.total}</div><div class="sub">Since May 1, 2026</div></div>
      <div class="sc b"><div class="lbl">Daily Avg</div><div class="val">${s.avg.toFixed(1)}</div><div class="sub">Excl. days off</div></div>
      <div class="sc a"><div class="lbl">Days Worked</div><div class="val">${s.days}</div><div class="sub">Days with output</div></div>
    </div>
    <div class="st">Weekly Breakdown – ${e}</div>
    <table><thead><tr><th>Week</th><th>Total</th><th>Days</th><th>Avg/Day</th></tr></thead><tbody>${rows}</tbody></table>`;
  }
}

// ── WEEKLY DETAIL ─────────────────────────────────────────
function renderWeekly() {
  const el = document.getElementById('p-weekly');
  const ws = filteredWeeks();
  const vs = cView==='team' ? STAFF : [cView];
  let html = '';
  ws.forEach(w=>{
    html += `<div class="st">${w.label}</div><table><thead><tr><th>Employee</th>`;
    w.days.forEach(d=>{ html+=`<th>${d}</th>`; });
    html += `<th>Total</th><th>Days</th><th>Avg/Day</th></tr></thead><tbody>`;
    vs.forEach(e=>{
      const arr = w.data[e] || w.days.map(()=>0);
      let tot=0, days=0;
      html += `<tr><td>${e}</td>`;
      arr.forEach(v=>{ tot+=v; if(v>0)days++; html+=`<td style="color:${v>0?'#e2e8f0':'#374f6b'}">${v>0?v:'—'}</td>`; });
      html += `<td style="font-weight:700;color:#0d9488">${tot}</td><td>${days}</td><td>${days>0?(tot/days).toFixed(1):'-'}</td></tr>`;
    });
    html += `</tbody></table><br>`;
  });
  el.innerHTML = html || '<div class="nodata">No data for selected period.</div>';
}

// ── LEADERBOARD ───────────────────────────────────────────
function renderLeaderboard() {
  const el = document.getElementById('p-leaderboard');
  const vs = cView==='team' ? STAFF : [cView];
  const sorted = [...vs].sort((a,b)=>empStats(b).total-empStats(a).total);
  const mx = empStats(sorted[0]).total || 1;
  const medals = ['🥇','🥈','🥉'];
  let rows = sorted.map((e,i)=>{
    const s=empStats(e), pct=Math.round(s.total/mx*100);
    return `<tr>
      <td style="font-weight:700;color:#0d9488">${i+1} ${medals[i]||''}</td>
      <td>${e}</td>
      <td style="font-weight:700;color:#0d9488">${s.total}</td>
      <td>${s.avg.toFixed(1)}</td>
      <td>${s.days}</td>
      <td><div style="background:#1e3a5f;border-radius:4px;height:8px;width:120px">
        <div style="background:#0d9488;height:8px;border-radius:4px;width:${pct}%"></div></div></td>
    </tr>`;
  }).join('');
  el.innerHTML = `<div class="st">Claims Modified Leaderboard</div>
  <table><thead><tr><th>Rank</th><th>Employee</th><th>Total</th><th>Avg/Day</th><th>Days</th><th>Progress</th></tr></thead>
  <tbody>${rows}</tbody></table>`;
}

// ── 90% THRESHOLD ─────────────────────────────────────────
function renderThreshold() {
  const el = document.getElementById('p-threshold');
  const vs = cView==='team' ? STAFF : [cView];
  const w1=WEEKS[0], w2=WEEKS[1];
  let rows = vs.map(e=>{
    const b=wTotal(w1,e), p2=wTotal(w2,e), tgt=Math.ceil(b*0.9);
    let pill, note;
    if (b===0) { pill=`<span class="pill am">N/A</span>`; note='No baseline output'; }
    else if (p2>=tgt) { pill=`<span class="pill g">✓ Met</span>`; note=`+${p2-tgt} above threshold`; }
    else { pill=`<span class="pill r">✗ Missed</span>`; note=`${tgt-p2} claims short`; }
    return `<tr><td>${e}</td><td>${b}</td><td>${p2}</td><td>${b>0?tgt:'—'}</td><td>${pill}</td><td style="font-size:12px;color:#64748b">${note}</td></tr>`;
  }).join('');
  el.innerHTML = `<div class="st">90% Threshold – Policy Effective May 2026</div>
  <p style="font-size:12px;color:#64748b;margin-bottom:12px">Baseline: May 4–8 &nbsp;|&nbsp; Comparison: May 11–16 &nbsp;|&nbsp;
  90%+ = <span style="color:#10b981">Met</span> &nbsp;|&nbsp; Below 90% = <span style="color:#ef4444">Missed</span></p>
  <table><thead><tr><th>Employee</th><th>Baseline (May 4–8)</th><th>Period 2 (May 11–16)</th><th>90% Target</th><th>Status</th><th>Notes</th></tr></thead>
  <tbody>${rows}</tbody></table>`;
}

// ── CHARTS ────────────────────────────────────────────────
function renderCharts() {
  const el = document.getElementById('p-charts');
  const vs = cView==='team' ? STAFF : [cView];
  ['ch-bar','ch-avg','ch-wow'].forEach(destroyChart);

  const totals = vs.map(e=>empStats(e).total);
  const avgs   = vs.map(e=>parseFloat(empStats(e).avg.toFixed(1)));
  const w1t    = vs.map(e=>wTotal(WEEKS[0],e));
  const w2t    = vs.map(e=>wTotal(WEEKS[1],e));
  const clrs   = ['#0d9488','#3b82f6','#8b5cf6','#f59e0b','#10b981','#ef4444','#06b6d4','#a855f7','#f97316'];
  const bh     = Math.max(280, vs.length*44);

  el.innerHTML = `

    <div style="position:relative;width:100%;height:${bh}px"><canvas id="ch-bar" role="img" aria-label="Total claims modified per employee"></canvas></div></div>
  <div class="cb"><h3>Daily average – active days only</h3>
    <div style="position:relative;width:100%;height:${bh}px"><canvas id="ch-avg" role="img" aria-label="Daily average claims per employee"></canvas></div></div>
  <div class="cb"><h3>Week-over-week comparison</h3>
    <div style="display:flex;gap:14px;margin-bottom:8px;font-size:12px;color:#94a3b8">
      <span><span style="display:inline-block;width:10px;height:10px;border-radius:2px;background:#0d9488;margin-right:4px"></span>May 4–8</span>
      <span><span style="display:inline-block;width:10px;height:10px;border-radius:2px;background:#3b82f6;margin-right:4px"></span>May 11–16</span>
    </div>
    <div style="position:relative;width:100%;height:${bh}px"><canvas id="ch-wow" role="img" aria-label="Week over week comparison"></canvas></div></div>`;

  // Use requestAnimationFrame to ensure canvases are painted before Chart.js touches them
  requestAnimationFrame(()=>{
    const c1=document.getElementById('ch-bar');
    const c2=document.getElementById('ch-avg');
    const c3=document.getElementById('ch-wow');
    if (!c1||!c2||!c3) return;

    const baseOpts = (data, lbl) => ({
      type:'bar',
      data:{ labels:vs, datasets:[{ label:lbl, data, backgroundColor:vs.map((_,i)=>clrs[i%clrs.length]), borderRadius:4 }] },
      options:{ indexAxis:'y', responsive:true, maintainAspectRatio:false,
        plugins:{ legend:{display:false} },
        scales:{ x:{ticks:{color:'#64748b'},grid:{color:'#1e3a5f'}}, y:{ticks:{color:'#94a3b8'},grid:{color:'#1e3a5f'}} } }
    });

    try { chartReg['ch-bar'] = new Chart(c1, baseOpts(totals,'Claims')); } catch(e){}
    try { chartReg['ch-avg'] = new Chart(c2, baseOpts(avgs,'Avg/Day')); } catch(e){}
    try {
      chartReg['ch-wow'] = new Chart(c3, {
        type:'bar',
        data:{ labels:vs, datasets:[
          { label:'May 4–8',   data:w1t, backgroundColor:'#0d9488', borderRadius:3 },
          { label:'May 11–16', data:w2t, backgroundColor:'#3b82f6', borderRadius:3 }
        ]},
        options:{ indexAxis:'y', responsive:true, maintainAspectRatio:false,
          plugins:{ legend:{display:false} },
          scales:{ x:{ticks:{color:'#64748b'},grid:{color:'#1e3a5f'}}, y:{ticks:{color:'#94a3b8'},grid:{color:'#1e3a5f'}} } }
      });
    } catch(e){}
  });
}
