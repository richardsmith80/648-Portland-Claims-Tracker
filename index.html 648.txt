// ── HELPERS ───────────────────────────────────────────────
function wTotal(w, e) { return (w.data[e] || []).reduce((a,b) => a+b, 0); }
function wDays(w, e)  { return (w.data[e] || []).filter(v => v > 0).length; }
function fWeeks()     { return cPeriod === 'YTD' ? WEEKS : WEEKS.filter(w => w.month === cPeriod); }
function empStats(e)  { let t=0,d=0; fWeeks().forEach(w=>{t+=wTotal(w,e);d+=wDays(w,e);}); return{total:t,days:d,avg:d>0?t/d:0}; }
function allStats(e)  { let t=0,d=0; WEEKS.forEach(w=>{t+=wTotal(w,e);d+=wDays(w,e);}); return{total:t,days:d,avg:d>0?t/d:0}; }
function sName(n)     { const p=n.split(' '); return p[0][0]+'. '+p[1]; }
function dc(id)       { if(chartReg[id]){try{chartReg[id].destroy();}catch(e){} delete chartReg[id];} }

function renderDataLog() {
  const el = document.getElementById('p-datalog');
  if (!el) return;
  const monthOrder=['Oct','Nov','Dec','Jan','Feb','Mar','Apr','May'];
  const byMonth={};
  WEEKS.forEach(w=>{if(!byMonth[w.month])byMonth[w.month]=[];byMonth[w.month].push(w);});
  let html=`<div class="alert">
    <strong>Duplicate protection active.</strong> ${WEEKS.length} unique weeks loaded. Any duplicate week IDs submitted are automatically discarded.
  </div>
  <div class="st">Coverage by month</div>`;
  monthOrder.forEach(mo=>{
    if(!byMonth[mo]) return;
    html+=`<div style="background:#0f1f3a;border:1px solid #1e3a5f;border-radius:10px;margin-bottom:12px;overflow:hidden">
      <div style="padding:10px 14px;background:#1e3a5f;font-size:12px;font-weight:600;color:#0d9488">${mo} 2025–2026 — ${byMonth[mo].length} week(s)</div>`;
    byMonth[mo].forEach(w=>{
      const tt=STAFF.reduce((a,e)=>a+wTotal(w,e),0);
      html+=`<div class="dl-row"><span class="dl-id">${w.id}</span><span class="dl-label">${w.label}</span><span style="color:#64748b;font-size:11px;margin-right:12px">${w.days.length} days</span><span class="dl-total">${tt.toLocaleString()}</span></div>`;
    });
    html+=`</div>`;
  });
  html+=`<div class="st">Known gaps</div>
  <div style="background:#0f1f3a;border:1px solid #1e3a5f;border-radius:10px;padding:12px 14px">
    ${KNOWN_GAPS.map(g=>`<div style="font-size:12px;color:#64748b;padding:4px 0;border-bottom:1px solid #1a2f4a">${g}</div>`).join('')}
    <div style="font-size:11px;color:#4a6080;padding-top:8px">Note: Feb 8–14 data has not been submitted yet. All other gaps are holidays or weekends.</div>
  </div>`;
  el.innerHTML=html;
}
